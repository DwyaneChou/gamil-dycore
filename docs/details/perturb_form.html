\left( \pd{\overline{\pi}}{\eta} + \pd{p^\prime}{\eta} \right) \pd{\eta}{\pi} \nabla_{\eta} \phi + \mathbf{F} \\ & = - \alpha \left( {\color{red}\nabla_\eta \overline{\pi}} + \nabla_\eta p^\prime \right) - \pd{\overline{\pi}}{\eta} \pd{\eta}{\pi} \nabla_{\eta} \phi - \pd{p^\prime}{\eta} \pd{\eta}{\pi} \nabla_{\eta} \phi + \mathbf{F} \\ & = - \alpha \left( {\color{red}\nabla_\eta \overline{\pi}} + \nabla_\eta p^\prime \right) - \pd{\overline{\pi}}{\overline{\phi}} \pd{\overline{\phi}}{\eta} \pd{\eta}{\pi} \nabla_{\eta} \phi - \pd{p^\prime}{\eta} \pd{\eta}{\pi} \nabla_{\eta} \phi + \mathbf{F} \\ & = - \alpha \left( {\color{red}\nabla_\eta \overline{\pi}} + \nabla_\eta p^\prime \right) + \frac{1}{\overline{\alpha}} \pd{\overline{\phi}}{\pi} \left( {\color{red}- \overline{\alpha} \nabla_\eta \overline{\pi}} + \nabla_\eta \phi^\prime \right) - \pd{p^\prime}{\eta} \pd{\eta}{\pi} \nabla_{\eta} \phi + \mathbf{F} \\ & = - \alpha \left( {\color{red}\nabla_\eta \overline{\pi}} + \nabla_\eta p^\prime \right) - \left( {\color{red}- \overline{\alpha} \nabla_\eta \overline{\pi}} + \nabla_\eta \phi^\prime \right) - \pd{p^\prime}{\eta} \pd{\eta}{\pi} \nabla_{\eta} \phi + \mathbf{F} \\ & = - \left( \alpha - \overline{\alpha} \right) \nabla_\eta \overline{\pi} - \alpha \nabla_\eta p^\prime - \nabla_\eta \phi^\prime - \pd{p^\prime}{\eta} \pd{\eta}{\pi} \nabla_{\eta} \phi + \mathbf{F} \\ & = - \alpha^\prime \nabla_\eta \overline{\pi} - \alpha \nabla_\eta p^\prime - \nabla_\eta \phi^\prime - \pd{p^\prime}{\eta} \pd{\eta}{\pi} \nabla_{\eta} \phi + \mathbf{F} \end{align*} \] 如果在静力气压坐标的基础上则扰动方程形式简化为 \[ \begin{align*} \dt{\mathbf{v}} & = - \frac{1}{\rho} \nabla_\pi p - \pd{p}{\pi} \nabla_\pi \phi + \mathbf{F} \\ & = - \frac{1}{\rho} \nabla_\pi p^\prime - \pd{p}{\pi} \nabla_\pi \phi^\prime + \mathbf{F} \\ \end{align*} \] 然后再进行广义地形追随坐标变换 \[ \begin{align*} \dt{\mathbf{v}} & = - \frac{1}{\rho} \nabla_\pi p^\prime - \pd{p}{\pi} \nabla_\pi \phi^\prime + \mathbf{F} \\ & = {\color{purple}- \frac{1}{\rho} \nabla_\eta p^\prime - \pd{p}{\eta} \pd{\eta}{\pi} \nabla_\eta \phi^\prime} + \mathbf{F} \end{align*} \]
