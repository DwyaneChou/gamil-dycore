静力气压坐标方程组

采用基于静力气压或质量坐标的预报方程组（Laprise 1992） \[ \begin{align} \dt{\mathbf{v}} & = - \frac{1}{\rho} \nabla_\pi p - \pd{p}{\pi} \nabla_\pi \phi + \mathbf{F} \\ {\color{red}\gamma \dt{w}} & {\color{red}= g \left( \pd{p}{\pi} - 1 \right) + \gamma F_z} \\ \dt{T} & = \frac{1}{C_p \rho} \dt{p} + \frac{Q}{C_p} \\ \hdiv + \pd{\dot{\pi}}{\pi} & = 0 \\ {\color{red}\dt{\phi}} & {\color{red}= w g} \quad \text{或} \quad {\color{blue}w = \frac{1}{g} \dt{\phi}} \end{align} \] 其中\(\pi\)为静力气压，地面有效静力气压\(\pi_{es} = \pi_s - \pi_t\)，\(\pi_s\)为静力地面气压，\(\pi_t\)为模式顶气压，\(\gamma = 0\)时表明采用静力平衡假设。上面的方程组中红色项在非静力模式里计算，蓝色项在静力模式里计算。

静力层结扣除

\[ \begin{align*} T(x, y, \pi, t) & = \overline{T}(\pi) + \t(x, y, \pi, t) \\ \phi(x, y, \pi, t) & = \overline{\phi}(\pi) + \phi^\prime(x, y, \pi, t) \\ p(x, y, \pi, t) & = \pi + p^\prime(x, y, \pi, t) \\ \pi_s(x, y, t) & = \overline{\pi}_s(x, y) + \pi_s^\prime(x, y, t) \end{align*} \]

\[ \begin{align*} \dt{\mathbf{v}} & = - \frac{1}{\rho} \nabla_\pi p^\prime - \pd{p}{\pi} \nabla_\pi \phi^\prime + \mathbf{F} \end{align*} \]

\[ \begin{align*} {\color{red}\gamma \dt{w}} & {\color{red}= g \pd{p^\prime}{\pi} + \gamma F_z} \end{align*} \]

\[ \begin{align*} \dt{\t} & = \frac{1}{C_p \rho} \dt{p^\prime} - \dot{\pi} \left( \d{\overline{T}}{\pi} - \frac{1}{C_p \rho} \right) + \frac{Q}{C_p} \\ & {\color{gray}= \frac{1}{C_p \rho} \dt{p^\prime} - \frac{\dot{\pi}}{\pi} \left( \d{\overline{T}}{\ln \pi} - \frac{R}{C_p} \frac{\pi}{\pi + p^\prime} \overline{T} - \frac{R}{C_p} \frac{\pi}{\pi + p^\prime} T^\prime \right) + \frac{Q}{C_p}} \\ \end{align*} \]

\[ \begin{align*} \dt{\phi^\prime} & = - \dot{\pi} \d{\overline{\phi}}{\pi} + w g \end{align*} \]

广义垂直坐标变换

气压梯度力 \[ \frac{1}{\rho} \nabla_\pi p^\prime = \frac{1}{\rho} \nabla_\eta p^\prime {\color{red}- \frac{1}{\rho} \pd{\eta}{\pi} \pde{p^\prime} \nabla_\eta \pi} \] \[ \pd{p}{\pi} \nabla_\pi \phi^\prime = \pd{p}{\pi} \nabla_\eta \phi^\prime {\color{blue}- \pd{p}{\pi} \pd{\eta}{\pi} \pd{\phi^\prime}{\eta} \nabla_\eta \pi} \] 两项合并 \[ \begin{align*} & \frac{1}{\rho} \nabla_\eta p^\prime + \pde{p} \pd{\eta}{\pi} \nabla_\eta \phi^\prime {\color{red}- \frac{1}{\rho} \pd{\eta}{\pi} \pde{p^\prime} \nabla_\eta \pi} {\color{blue}- \pd{p}{\pi} \pd{\eta}{\pi} \pd{\phi^\prime}{\eta} \nabla_\eta \pi} \\ = & \frac{1}{\rho} \nabla_\eta p^\prime + \pde{p} \pd{\eta}{\pi} \nabla_\eta \phi^\prime - \pd{\eta}{\pi} \nabla_\eta \pi \left( \frac{1}{\rho} \pde{p^\prime} + \pd{p}{\pi} {\color{magenta}\pde{\phi^\prime}} \right) \\ = & \frac{1}{\rho} \nabla_\eta p^\prime + \pde{p} \pd{\eta}{\pi} \nabla_\eta \phi^\prime - \pd{\eta}{\pi} \nabla_\eta \pi \left( \frac{1}{\rho} \pde{p^\prime} + \pd{p}{\pi} {\color{magenta}\pde{\phi}} - \pd{p}{\pi} {\color{magenta}\d{\overline{\phi}}{\eta}} \right) \\ = & \frac{1}{\rho} \nabla_\eta p^\prime + \pde{p} \pd{\eta}{\pi} \nabla_\eta \phi^\prime - \pd{\eta}{\pi} \nabla_\eta \pi \left( \frac{1}{\rho} \pde{p^\prime} - \pd{p}{\pi} \frac{1}{\rho} \pde{\pi} - \pd{p}{\pi} \d{\overline{\phi}}{\eta} \right) \\ = & \frac{1}{\rho} \nabla_\eta p^\prime + \pde{p} \pd{\eta}{\pi} \nabla_\eta \phi^\prime - \pd{\eta}{\pi} \nabla_\eta \pi \left( \frac{1}{\rho} \pde{p^\prime} - \frac{1}{\rho} \pde{p} - \pd{p}{\pi} \d{\overline{\phi}}{\eta} \right) \\ = & \frac{1}{\rho} \nabla_\eta p^\prime + \pde{p} \pd{\eta}{\pi} \nabla_\eta \phi^\prime - \pd{\eta}{\pi} \nabla_\eta \pi \left( - \frac{1}{\rho} \pde{\pi} - \pd{p}{\pi} \d{\overline{\phi}}{\eta} \right) \\ = & \frac{1}{\rho} \nabla_\eta p^\prime + \pde{p} \pd{\eta}{\pi} \nabla_\eta \phi^\prime + \nabla_\eta \pi \left( \frac{1}{\rho} + \pd{p}{\pi} \d{\overline{\phi}}{\pi} \right) \\ = & \frac{1}{\rho} \nabla_\eta p^\prime + \pde{p} \pd{\eta}{\pi} \nabla_\eta \phi^\prime + \nabla_\eta \pi \left( \frac{1}{\rho} - \pd{p}{\pi} \frac{1}{\overline{\rho}} \right) \\ = & \frac{1}{\rho} \nabla_\eta p^\prime + \pde{p} \pd{\eta}{\pi} \nabla_\eta \phi^\prime + \nabla_\eta \pi \left( \frac{1}{\rho} - \pd{p}{\pi} \frac{R \overline{T}}{\pi} \right) \\ \end{align*} \]

\[ \begin{align*} \dt{\mathbf{v}} & = {\color{red}- \frac{1}{\rho} \nabla_\eta p^\prime} - {\color{magenta}\pde{p} \pd{\eta}{\pi}} \nabla_\eta \phi^\prime - \left( \frac{1}{\rho} - {\color{magenta}\pde{p} \pd{\eta}{\pi}} \frac{R \overline{T}}{\pi} \right) \nabla_\eta \pi + \mathbf{F} \\ {\color{red}\gamma \dt{w}} & {\color{red}= g \pd{p^\prime}{\eta} \pd{\eta}{\pi} + \gamma F_z} \\ \dt{\t} & = {\color{red}\frac{1}{C_p \rho} \dt{p^\prime}} - \po \dot{\eta} \left( \d{\overline{T}}{\pi} - \frac{1}{C_p \rho} \right) + \frac{Q}{C_p} \\ \pdt{} \po & = - \pd{}{x} \left( u \po \right) - \pd{}{y} \left( v \po \right) - \pde{} \left( \de \po \right) \\ \dt{\phi^\prime} & = - \po \dot{\eta} \d{\overline{\phi}}{\pi} + w g \end{align*} \]

\[ \begin{align} \pdt{u} & = - u \pd{u}{x} - v \pd{u}{y} - \de \pde{u} - \frac{1}{\rho} \pdx{\pi} - \pdx{\phi} - f^* v \\ \pdt{v} & = - u \pd{v}{x} - v \pd{v}{y} - \de \pde{v} - \frac{1}{\rho} \pdy{\pi} - \pdy{\phi} + f^* u \\ \pdt{T} & = - u \pd{T}{x} - v \pd{T}{y} - \de \pde{T} + \frac{1}{C_p \rho} \dt{\pi} \\ \pdt{} \po & = - \pd{}{x} \left( u \po \right) - \pd{}{y} \left( v \po \right) - \pde{} \left( \de \po \right) \\ \pd{\phi}{p} & = - \frac{R T}{\pi} \end{align} \]
